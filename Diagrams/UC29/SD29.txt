@startuml
skinparam sequenceMessageAlign center
skinparam monochrome true
title UC-29: Xem danh sách đơn hàng trạng thái "Pending"

actor "Nhân viên Sales" as Staff
boundary "OrderMgmt_UI" as UI
control "Order_Controller" as Ctrl
database "DB" as DB

Staff -> UI: 1. Chọn mục quản lý đơn hàng
activate Staff
activate UI

UI -> Ctrl: 2. Yêu cầu tải danh sách đơn hàng "Pending"
activate Ctrl

Ctrl -> DB: 3. getPendingOrders()
activate DB

alt [x Không có đơn hàng chờ xử lý]
    DB --> Ctrl: 4. Phản hồi danh sách rỗng
    Ctrl --> UI: 5. Trả về trạng thái không có dữ liệu
    UI --> Staff: 6. Hiển thị bảng trống và thông báo
else [x Có đơn hàng chờ xử lý]
    DB --> Ctrl: 7. Phản hồi danh sách đơn hàng hợp lệ
    Ctrl --> UI: 8. Trả về dữ liệu danh sách đơn hàng
    UI --> Staff: 9. Hiển thị danh sách đơn hàng dưới dạng bảng
end

deactivate DB
deactivate Ctrl
deactivate UI
deactivate Staff
@enduml
