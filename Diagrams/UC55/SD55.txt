@startuml
skinparam defaultFontColor black
skinparam monochrome true
title UC-55: Cấu hình phí giao hàng

actor "Business Admin" as Admin
boundary Shipping_UI
control Shipping_Controller
database DB

activate Admin
activate Shipping_UI
Admin -> Shipping_UI: 1. Xem danh sách các khu vực vận chuyển
activate Shipping_Controller
Shipping_UI -> Shipping_Controller: 2. Gửi yêu cầu lấy danh sách khu vực
activate DB
Shipping_Controller -> DB: 3. getShippingRegions()
DB --> Shipping_Controller: 4. Trả về danh sách khu vực và phí hiện tại
Shipping_UI --> Admin: 5. Hiển thị danh sách phí vận chuyển
Admin -> Shipping_UI: 6. Nhập mức phí mới và nhấn lưu
Shipping_UI -> Shipping_Controller: 7. Gửi cấu hình phí mới
Shipping_Controller -> DB: 8. checkShippingFee(input)
DB --> Shipping_Controller: 9. Kết quả xác thực dữ liệu

alt [x Giá trị âm hoặc sai định dạng chữ]
    Shipping_Controller --> Shipping_UI: 10. Báo lỗi dữ liệu không hợp lệ
    Shipping_UI --> Admin: 11. Hiển thị thông báo dữ liệu không hợp lệ
else [x Dữ liệu hợp lệ]
    Shipping_Controller -> DB: 12. updateShippingFees(data)
    DB --> Shipping_Controller: 13. success
    deactivate DB
    Shipping_Controller --> Shipping_UI: 14. Báo lỗi cập nhật thành công
    deactivate Shipping_Controller
    Shipping_UI --> Admin: 15. Hiển thị thông báo cập nhật thành công
    deactivate Admin
    deactivate Shipping_UI
end

@enduml
