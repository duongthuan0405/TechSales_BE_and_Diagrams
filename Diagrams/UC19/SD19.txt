@startuml
skinparam sequenceMessageAlign center
skinparam monochrome true
title UC-19: Chọn phương thức thanh toán

actor "Khách hàng" as User
boundary "Checkout_UI" as UI
control "Checkout_Controller" as Ctrl
database "DB" as DB

activate User
activate UI
User -> UI: 1. Nhấn chọn phương thức thanh toán

activate Ctrl
UI -> Ctrl: 2. Yêu cầu cập nhật phương thức đã chọn

activate DB
Ctrl -> DB: 3. getPaymentInstruction(methodID)

alt [x Phương thức thanh toán khả dụng]
    DB --> Ctrl: 4. Phản hồi thông tin hướng dẫn chi tiết
    Ctrl --> UI: 5. Trả về thông tin hiển thị và lưu phiên làm việc
    UI --> User: 6. Hiển thị hướng dẫn thanh toán cho người dùng
else [x Lỗi tải thông tin phương thức]
    DB --> Ctrl: 7. Phản hồi lỗi truy xuất dữ liệu
    deactivate DB
    Ctrl --> UI: 8. Thông báo lỗi hệ thống tạm thời
    deactivate Ctrl
    UI --> User: 9. Hiển thị yêu cầu chọn lại phương thức
    deactivate User
    deactivate UI
end

@enduml
