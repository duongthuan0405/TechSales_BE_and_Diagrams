@startuml
skinparam sequenceMessageAlign center
skinparam monochrome true
title UC-23: Xem chi tiết đơn hàng cũ

actor "Khách hàng" as User
boundary "OrderHistory_UI" as UI
control "Order_Controller" as Ctrl
database "DB" as DB


activate User
activate UI
User -> UI: 1. Nhấn xem chi tiết đơn hàng
activate Ctrl
UI -> Ctrl: 2. Yêu cầu tải dữ liệu chi tiết đơn hàng
activate DB
Ctrl -> DB: 3. getOrderDetails(orderID)

alt [x Dữ liệu không tồn tại hoặc lỗi truy xuất]
    DB --> Ctrl: 4. Phản hồi lỗi không tìm thấy dữ liệu
    Ctrl --> UI: 5. Trả về thông báo lỗi hệ thống
    UI --> User: 6. Hiển thị cảnh báo lỗi và quay lại danh sách
else [x Truy xuất dữ liệu thành công]
    DB --> Ctrl: 7. Phản hồi chi tiết bản ghi đơn hàng
    deactivate DB
    Ctrl --> UI: 8. Trả về dữ liệu hiển thị giao diện
    deactivate Ctrl
    UI --> User: 9. Hiển thị giao diện chi tiết hóa đơn lịch sử
    deactivate User
    deactivate UI
end

@enduml
