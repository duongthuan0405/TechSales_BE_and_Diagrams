@startuml
title UC-36: Tìm kiếm đơn hàng

actor "Nhân viên Sales" as Staff
boundary OrderManagement_UI as UI
control Order_Controller as Ctrl
database DB


activate Staff
activate UI
Staff -> UI : 1. Nhập từ khóa (ID/SĐT) và nhấn tìm kiếm
activate Ctrl
UI -> Ctrl : 2. Yêu cầu tìm kiếm đơn hàng(keyword)
activate DB
Ctrl -> DB : 3. searchOrders(keyword)

alt x Tìm thấy đơn hàng khớp
    DB -> Ctrl : 4. danh sách đơn hàng
    Ctrl -> UI : 5. Trả về danh sách kết quả
    UI -> Staff : 6. Hiển thị danh sách đơn hàng lên bảng
else x Không tìm thấy kết quả
    DB -> Ctrl : 7. null / empty
    deactivate DB
    Ctrl -> UI : 8. Thông báo không tìm thấy kết quả
    deactivate Ctrl
    UI -> Staff : 9. Hiển thị thông báo lỗi tìm kiếm
    deactivate Staff
    deactivate UI
end

@enduml
