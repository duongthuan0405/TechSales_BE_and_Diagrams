@startuml
skinparam sequenceMessageAlign center
skinparam monochrome true
title UC-25: Hủy đơn hàng trạng thái "Pending"

actor "Khách hàng" as User
boundary "OrderDetail_UI" as UI
control "Order_Controller" as Ctrl
database "DB" as DB

activate User
activate UI
User -> UI: 1. Nhấn nút hủy đơn hàng

UI --> User: 2. Hiển thị hộp thoại yêu cầu xác nhận

alt [x Khách hàng xác nhận đồng ý]
    User -> UI: 3. Nhấn nút xác nhận hủy
    activate Ctrl
    UI -> Ctrl: 4. Gửi yêu cầu hủy đơn hàng
    
    activate DB
    Ctrl -> DB: 5. checkOrderStatus(orderID)
    
    alt [x Trạng thái đã thay đổi sang Approved]
        DB --> Ctrl: 6. Phản hồi trạng thái không còn là Pending
        Ctrl --> UI: 7. Báo lỗi đơn hàng không thể tự hủy
        UI --> User: 8. Hiển thị thông báo và yêu cầu làm mới trang
    else [x Trạng thái vẫn là Pending]
        DB --> Ctrl: 9. Xác nhận trạng thái hợp lệ
        
        Ctrl -> DB: 10. updateStatusToCanceled(orderID)
        Ctrl -> DB: 11. returnStockQuantity(items)
        DB --> Ctrl: 12. Xác nhận cập nhật thành công
        deactivate DB
        
        Ctrl --> UI: 13. Phản hồi hủy đơn thành công
        deactivate Ctrl
        UI --> User: 14. Thông báo thành công và cập nhật lại giao diện
    end
else [x Khách hàng từ chối]
    User -> UI: 15. Nhấn nút đóng hoặc từ chối
    UI --> User: 16. Đóng hộp thoại xác nhận
    deactivate User
    deactivate UI
end

@enduml
