@startuml
skinparam sequenceMessageAlign center
skinparam monochrome true
title UC-18: Nhập mã giảm giá

actor "Khách hàng" as User
boundary "Checkout_UI" as UI
control "Voucher_Controller" as Ctrl
database "DB" as DB

activate User
activate UI
User -> UI: 1. Nhập mã voucher và nhấn áp dụng

activate Ctrl
UI -> Ctrl: 2. Yêu cầu kiểm tra mã giảm giá

activate DB
Ctrl -> DB: 3. getVoucherDetails(voucherCode)

DB --> Ctrl: 4. Phản hồi thông tin và điều kiện voucher
deactivate DB

alt [x Mã không hợp lệ hoặc không đủ điều kiện]
    Ctrl --> UI: 5. Phản hồi lỗi mã giảm giá
    UI --> User: 6. Hiển thị thông báo lỗi và giữ nguyên tổng tiền
else [x Mã hợp lệ]
    Ctrl -> Ctrl: 7. calculateDiscount(orderValue, voucherData)
    Ctrl --> UI: 8. Phản hồi số tiền giảm và tổng tiền mới
    deactivate Ctrl
    UI --> User: 9. Hiển thị thông báo thành công và cập nhật hóa đơn
    deactivate User
    deactivate UI
end

@enduml
