@startuml
skinparam sequenceMessageAlign center
skinparam monochrome true
title UC-20: Xác nhận mua hàng

actor "Khách hàng" as User
boundary "Checkout_UI" as UI
control "Order_Controller" as Ctrl
database "DB" as DB

activate User
activate UI
User -> UI: 1. Nhấn nút xác nhận đặt hàng

activate Ctrl
UI -> Ctrl: 2. Yêu cầu tạo đơn hàng

note over Ctrl: Bắt đầu giao dịch nguyên tử (Atomic)

activate DB
Ctrl -> DB: 3. checkAndReserveStock(items)

alt [x Một hoặc nhiều sản phẩm bị thiếu hàng]
    DB --> Ctrl: 4. Phản hồi lỗi thiếu tồn kho
    Ctrl --> UI: 5. Thông báo lỗi số lượng không đủ
    UI --> User: 6. Hiển thị cảnh báo yêu cầu sửa giỏ hàng
else [x Đủ hàng tồn kho]
    DB --> Ctrl: 7. Xác nhận giữ hàng thành công
    
    Ctrl -> DB: 8. createOrder(orderInfo)
    Ctrl -> DB: 9. clearPurchasedItemsFromCart(userID)
    
    alt [x Có lỗi xảy ra trong quá trình xử lý]
        Ctrl -> DB: 10. Thực hiện Rollback()
        Ctrl --> UI: 11. Phản hồi lỗi xử lý giao dịch
        UI --> User: 12. Hiển thị thông báo sự cố hệ thống
    else [x Thực hiện thành công toàn bộ]
        Ctrl -> DB: 13. Thực hiện Commit()
        DB --> Ctrl: 14. Xác nhận lưu dữ liệu vĩnh viễn
        deactivate DB
        Ctrl --> UI: 15. Phản hồi đặt hàng thành công
        deactivate Ctrl
        UI --> User: 16. Thông báo thành công và điều hướng trang
        deactivate User
        deactivate UI
        note over Ctrl: Kích hoạt tự động UC-21
    end
end

@enduml
